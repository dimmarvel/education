class Distance
{
  private:
    const float MTF;
    int feet;
    float inches;
  public:
    Distance ( ) : feet ( 0 ), inches ( 0.0 ), MTF ( 3.280833F )
      { }
    explicit Distance ( float meters ) : MTF ( 3.280833F )
      {
        float fltfeet = MTF * meters;
        feet = int ( fltfeet );
        inches = 12 * ( fltfeet – feet );
      }
    void showdist ( )
      { cout << feet << "\'-" << inches << '\"'; }
};
